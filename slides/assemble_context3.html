"this" expression

This expression in any context to refer to the current context.

src/data/people.json

[
  "Jon Schlinkert",
  "Brian Woodward" 
]

src/templates/pages/people.html

<ul> 
  {{#each people }} 
  <li> {{this }} </li > 
  {{/each }} 
</ul>

<ul>
  <li>Jon Schlinkert</li>
  <li>Brian Woodward</li>
</ul>
